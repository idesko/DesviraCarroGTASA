{$CLEO .cs}
{$NOSOURCE}
0000:

// RC Pilot Test
// Press T to spawn a toy RC plane with a pedestrian inside.
// This is a safe visual test for the RC cutscene swap.

0@ = -1 // RC vehicle
1@ = -1 // pilot ped
TimerA = 2500

:MAIN
wait 0

if
    0256: player $PLAYER_CHAR defined
then
    if
        056D: actor $PLAYER_ACTOR defined
    then
        if
            856D: not actor 1@ defined
        then
            if and
                0AB0: key_pressed 84
                TimerA > 2000
            then
                gosub @SPAWN_RC_WITH_PILOT
            end
        else
            if
                0118: actor 1@ dead
            then
                gosub @CLEANUP
            end
        end
    end
end

goto @MAIN

:SPAWN_RC_WITH_PILOT
TimerA = 0

// Load the toy RC and the pilot model first.
0247: request_model 464 // rcbaron
0247: request_model 7 // male01

:WAIT_MODELS
wait 0
if and
    0248: model 464 available
    0248: model 7 available
then
    jump @MODEL_READY
end
goto @WAIT_MODELS

:MODEL_READY
// Spawn a little in front of CJ so the test is easy to see.
04C4: create_coordinate 2@ 3@ 4@ from_actor $PLAYER_ACTOR offset 0.0 6.0 1.0
00A5: 0@ = create_car 464 at 2@ 3@ 4@
009A: 1@ = create_actor 4 model 7 at 2@ 3@ 4@

// Put the pilot in the driver's seat so the RC looks occupied.
072A: put_actor 1@ into_car 0@ driverseat

// Make sure the pilot stays put for the test.
0350: set_char_stay_in_same_place 1@ to 1
0568: set_char_never_targetted 1@ untargetable 1

0249: release_model 464
0249: release_model 7
return

:CLEANUP
if
    056D: actor 1@ defined
then
    009B: destroy_actor 1@
end
if
    056E: car 0@ defined
then
    00A6: destroy_car 0@
end
0@ = -1
1@ = -1
return
